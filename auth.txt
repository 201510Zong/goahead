#   Catch-all route without authentication for all URIs
#
uri: none: none: /: :

#
#       uri: example.com: none: /auth/post/login.html: : referrer:/auth/post/index.html
#       uri: example.com: none: /form/login: : referrer:/auth/post/index.html
